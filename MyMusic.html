            import requests

history = pd.read_csv('myHystory.csv')
            
#Desde dónde empezar a buscar
            lastDataAvailable = history.iloc[0].date
            newSearchFrom = lastDataAvailable + 1
            
ApiKey = 'YOUR_API_KEY'
            
#páginas a hacer request
            extra = f'limit=200&from={newSearchFrom}&format=json'
            url = f"http://ws.audioscrobbler.com/2.0/?method=user.getrecenttracks&user=LauShang&api_key={ApiKey}&{extra}"
            json_ = requests.get(url).json()
            totalPages = int(json_['recenttracks']['@attr']['totalPages'])
            
#Get new data
            progress = {'num':[round(i*totalPages/10) for i in range(1,11)],'per':[f'{i*10}%' for i in range(1,11)]}
            pages = range(1,totalPages+1)
            old = pd.DataFrame({})
            for f in pages:
                extra = f'limit=200&from={newSearchFrom}&page={f}&format=json'
                url = f"http://ws.audioscrobbler.com/2.0/?method=user.getrecenttracks&user=LauShang&api_key={ApiKey}&{extra}"
                l = requests.get(url).json()
                my = {'artist':[], 'album':[], 'date':[], 'name':[], 'dateMexicoCity':[]}
                for x in l['recenttracks']['track']:
                    a = x['artist']['#text']
                    al = x['album']['#text']
                    d = x['date']['uts']
                    n = x['name']
                    my['artist'].append(a)
                    my['album'].append(al)
                    my['date'].append(d)
                    my['name'].append(n)
                    my['dateMexicoCity'].append(datetime.fromtimestamp(int(d)))
                new = pd.DataFrame(my)
                old = old.append(new,ignore_index = True)
                if f in progress['num']:
                    i = progress['num'].index(f)
                    print(progress['per'][i])
            
#juntar DataFrames
            newdf = old.append(history,ignore_index = True)
            
#Exportar
            newdf.to_csv('myHystory.csv', index=False)
